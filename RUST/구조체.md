구조체 선언 

```rust
struct Member {
    age: i32,
    active: bool,
    name: String
}
```

```rust
fn 구조체_사용하기() {
    // 1. 기본적인 구조체 인스턴스 생성
    let mut mem = Member {
        age: 32,
        active: true,
        name: String::from("nys")
    };

    // 2. 인스턴스 업데이트
    mem.age = 18;

    // 3. 기존 인스턴스로 새 인스턴스 생성  
    let mut mem2 = Member {
        age: 33,
        ..mem
    };

    println!("{}", mem.age);
    println!("{}", mem2.age);
    // 3. 과정에서 소유권이 이동했다.
    // ERROR
    println!("{}", mem.name);
    println!("{}", mem2.name);
}
```
