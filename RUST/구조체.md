구조체 선언 

```rust
struct Member {
    age: i32,
    active: bool,
    name: String
}
```

```rust
fn 구조체_immutable() {
    // 1. 기본적인 구조체 인스턴스 생성
    let mem = Member {
        age: 32,
        active: true,
        name: String::from("nys")
    };

    // mem변수를 mut로 지정하지 않았기 때문에 아래 두 구문은 컴파일 에러를 발생시킨다.
    // ERROR
    mem.age = 32;
    mem.name.push_str("!!");
}
```

```rust
fn 구조체_인스턴스_업데이트() {
    // 1. 기본적인 구조체 인스턴스 생성
    let mut mem = Member {
        age: 32,
        active: true,
        name: String::from("nys")
    };

    // 2. 인스턴스 업데이트
    mem.age = 18;

    // 3. 기존 인스턴스로 새 인스턴스 생성  
    // 이건 자바스크립트 문법과 비슷한듯?
    let mut mem2 = Member {
        age: 33,
        ..mem
    };

    println!("{}", mem.age);
    println!("{}", mem2.age);
    // ERROR
    println!("{}", mem.name); // 3. 과정에서 소유권이 이동했다.
    println!("{}", mem2.name);
}
```
