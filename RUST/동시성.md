fn thread_기본적인_사용법() {
    let t1join = thread::spawn(|| {
        for i in 1..5 {
            println!("thread1 {} ",i);
            thread::sleep(Duration::from_millis(1));
        }
        
    });

    let t2join = thread::spawn(|| {
        for i in 1..5 {
            println!("thread2 {} ",i);

        }

}
