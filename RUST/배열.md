
# 배열의 선언과 반복문 사용
```rust
    // [타입; 배열크기] 형태로 배열 타입을 선언한다.
    let mut arr: [i32; 5] = [1, 2, 3, 4, 5];

    println!("{}", arr.len());

    // arr은 참조로 전달된다.
    for element in &arr {
        println!("{}", element);
    }
    
    // WHY:
    // [참조로 전달하지 않을수도 있지만 
    // 이 경우 소유권 규칙에 따라 arr은 유효하지 않게된다.]
    // -> 라고 생각했지만 잘된다..
    for element in arr {
        println!("{}", element);
    }
    println!("{}", arr.len());
```
# 초기화 편의문법
```rust
// [초기값; 갯수]로 데이터를 할당하면 모든값이 [초기값]으로 초기화된 [갯수]크기의 배열을 생성한다.
// 아래는 모든 원소의 값이 3인 5크기의 배열을 생성하는것이다.  
let a: [i32; 5] = [3; 5];
```

# WHY: 반복문에서 &붙이는것과 안붙이는것의 차이 
```rust
    let mut arr: [i32; 5] = [1, 2, 3, 4, 5];

    let mut i = 0;
    // 배열의 각 원소를 순회하며 각 원소의 값을 기존값 + 10으로 바꿔주는 처리
    // 이때 arr을 &arr으로 바꾸면 에러가 발생한다.
    for element in arr {
        arr[i] = element + 10;
        i = i + 1;
    }
    
    println!("{}", arr[2]);
    println!("end");
```
